function normSt(v){if(!v)return'bestelling';var s=v.toLowerCase().trim();if(s.indexOf('bestell')>-1)return'bestelling';if(s.indexOf('onderweg')>-1)return'onderweg';if(s.indexOf('binnen')>-1)return'binnen';if(s.indexOf('verkocht')>-1)return'verkocht';return'bestelling';}

function spL(line,sep){var r=[],cur='',inq=false;for(var i=0;i<line.length;i++){var ch=line[i];if(ch==='"'){inq=!inq;}else if(ch===sep&&!inq){r.push(cur);cur='';}else{cur+=ch;}}r.push(cur);return r;}

function parseCSV(txt){
var fl=txt.split('\n')[0];
var sep=(fl.match(/;/g)||[]).length>(fl.match(/,/g)||[]).length?';':',';
var lines=txt.split('\n').map(function(l){return l.replace(/\r/,'');});
if(lines.length<2)return[];
var hdrs=spL(lines[0],sep).map(function(h){return h.replace(/^"|"$/g,'').trim().toLowerCase();});
var fm=hdrs.map(function(h){return CMAP[h]||null;});
var rows=[];
for(var i=1;i<lines.length;i++){
var line=lines[i].trim();if(!line)continue;
var cells=spL(line,sep).map(function(cell){return cell.replace(/^"|"$/g,'').trim();});
var obj={id:mkid()};
fm.forEach(function(field,idx){
if(field&&cells[idx]!==undefined){
var val=cells[idx];
if(field==='status')val=normSt(val);
if(field==='km')val=val.replace(/\./g,'').replace(/,/g,'').replace(/\s/g,'');
if(field==='inkoop'||field==='verkoop'||field==='resultaat'){
val=val.replace(/\u20ac/g,'').replace(/EUR/g,'').replace(/\s/g,'');
var parts=val.split(',');
if(parts.length===2&&parts[1].length<=2){val=parts[0].replace(/\./g,'')+'.'+parts[1];}
else{val=val.replace(/\./g,'').replace(',','.');}
val=val.replace(/[^0-9.-]/g,'');
}
obj[field]=val||null;
}
});
if(obj.kenteken||obj.vin)rows.push(obj);
}
return rows;
}

function publishToAutotelex(eid) {
var id = decodeURIComponent(eid);
var c = null;
for (var i = 0; i < cars.length; i++) { if (cars[i].id === id) { c = cars[i]; break; } }
if (!c) return;

if (!c.kenteken) {
alert('Vul eerst een kenteken in voordat je naar Autotelex publiceert.');
return;
}

if (!c.btw_marge) {
alert('Vul eerst in of het een BTW of Marge auto is (bewerk de auto).');
return;
}

if (!confirm('Auto ' + c.kenteken + ' publiceren naar AutotelexADS?')) return;

showToast('Bezig met publiceren naar Autotelex...');

// Step 1: Create vehicle by license plate
adsApiFetch('POST', '/api/v2/vehicle/create-by-licenseplate', {
licensePlate: c.kenteken.replace(/[-\s]/g, '')
}).then(function(data) {
var stockId = data.adsStockId || data.stockId || data.id;
if (!stockId) throw new Error('Geen stockId ontvangen van Autotelex');

var promises = [];

// Step 2: Set price if available
if (c.verkoop) {
promises.push(adsApiFetch('PUT', '/api/v2/advertisement/' + stockId + '/price', {
price: parseFloat(c.verkoop)
}));
}

// Step 3: Set BTW/Marge
var isVat = c.btw_marge === 'btw' ? 'true' : 'false';
promises.push(adsApiFetch('PUT', '/api/v2/vehicle/' + stockId + '/vat-margin/' + isVat, null));

// Step 4: Set status
if (c.status === 'binnen') {
promises.push(adsApiFetch('PUT', '/api/v2/stock/' + stockId + '/set-in-stock', null));
} else if (c.status === 'bestelling' || c.status === 'onderweg') {
// Set as expected with date if available
if (c.datum) {
var dp = c.datum.toString().split('-');
var dateObj;
if (dp.length === 3 && dp[2].length === 4) {
dateObj = new Date(dp[2], dp[1]-1, dp[0]);
} else {
dateObj = new Date(c.datum);
}
var ts = Math.floor(dateObj.getTime() / 1000);
promises.push(adsApiFetch('PUT', '/api/v2/stock/' + stockId + '/set-expected/' + ts, null));
}
}

return Promise.all(promises).then(function() {
// Save stockId to car record
apiFetch('PATCH', 'autos?id=eq.' + id, {ads_stock_id: String(stockId)}).then(function(){
for (var i=0; i<cars.length; i++) {
if (cars[i].id === id) { cars[i].ads_stock_id = String(stockId); break; }
}
});
showToast('Gepubliceerd naar AutotelexADS! StockId: ' + stockId);
});
}).catch(function(e) {
showToast('Fout: ' + (e.message || 'Kon niet publiceren naar Autotelex'));
console.error('Autotelex API error:', e);
});
}
